## 函数闭包

### 函数闭包的举例应用
 - 生成一个函数数组，分别返回 1-n 
     ```
     //类似JavaScript高级程序设计 7.2 闭包 中的一个例子
     
     //错误示例
     function n(num){
    	var arr = new Array(num);
    	for(var i=1;i<=num;i++){
    		arr[i]=function(){
    			return i;
    		}
    	}
    	return arr;
    }
    
    n(5)[0]//5
    n(5)[5]//5
    
    //正确示例
     function n(num){
    	var arr = new Array(num);
    	for(var i=1;i<=num;i++){
    		arr[i]=function(thisI){
    			return function(){
    			    return thisI;
    			}
    		}(i);
    	}
    	return arr;
    }
    
    n(3)[0]()//1
    n(3)[1]()//2
    n(3)[2]()//3
     ```
